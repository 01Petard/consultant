// mysql

create database  if  not exists volunteer;
use volunteer;
create table if not exists reservation
(
    id                 bigint  primary key auto_increment   not null comment '主键ID',
    name               varchar(50) not null comment '考生姓名',
    gender             varchar(2)  not null comment '考生性别',
    phone              varchar(20) not null comment '考生手机号',
    communication_time datetime    not null comment '沟通时间',
    province           varchar(32) not null comment '考生所处的省份',
    estimated_score    int         not null comment '考生预估分数'
)


// pg
CREATE DATABASE volunteer;

\c volunteer;

CREATE EXTENSION IF NOT EXISTS vector;

CREATE TABLE IF NOT EXISTS reservation (
    id BIGSERIAL PRIMARY KEY NOT NULL,
    name VARCHAR(50) NOT NULL,
    gender VARCHAR(2) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    communication_time TIMESTAMP NOT NULL,
    province VARCHAR(32) NOT NULL,
    estimated_score INT NOT NULL,
    embedding vector(1536) -- pgvector 向量字段
);

COMMENT ON COLUMN reservation.id IS '主键ID';
COMMENT ON COLUMN reservation.name IS '考生姓名';
COMMENT ON COLUMN reservation.gender IS '考生性别';
COMMENT ON COLUMN reservation.phone IS '考生手机号';
COMMENT ON COLUMN reservation.communication_time IS '沟通时间';
COMMENT ON COLUMN reservation.province IS '考生所处的省份';
COMMENT ON COLUMN reservation.estimated_score IS '考生预估分数';
COMMENT ON COLUMN reservation.embedding IS '向量表示字段';
